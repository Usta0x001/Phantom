=== auto_queue_cve_exploits ===
<tool name="auto_queue_cve_exploits">
    <description>Automatically queue CVE-based exploitation hypotheses from fingerprinted tech stack.
      
      This is the MAIN INTEGRATION FUNCTION that bridges:      
      WORKFLOW:
      1. Parse tech_stack dictionary for products with versions
      2. Call version_to_cves() for each technology
      3. Map CVEs to vulnerability classes (sqli, xss, rce, path_traversal, etc.)
      4. Generate attack surfaces based on product/CVE combination
      5. Queue hypotheses in hypothesis ledger
      6. Return prioritized exploitation plan      
      ELITE FEATURE:</description>
    
    <parameters>
      <parameter name="tech_stack" type="dict" required="true">
        <description>Tech stack dictionary from identify_tech_stack().</description>
      </parameter>
      
      <parameter name="base_url" type="string" required="true">
        <description>Target base URL for generating attack surfaces.</description>
      </parameter>
      
      <parameter name="hypothesis_ledger" type="object" required="false">
        <description>HypothesisLedger instance for direct integration.</description>
      </parameter>
      
      <parameter name="min_severity" type="string" required="false" default="MEDIUM">
        <description>Minimum CVE severity to queue.</description>
      </parameter>
      
      <parameter name="prioritize_exploits" type="boolean" required="false" default="true">
        <description>If true, prioritize CVEs with available exploits (Metasploit, PoC).</description>
      </parameter>
      
      <parameter name="max_hypotheses" type="integer" required="false" default="20">
        <description>Maximum number of hypotheses to queue.</description>
      </parameter>
    </parameters>
    
    <returns>
      <description>
        Dictionary with exploitation plan and queuing summary:
        {
          "hypotheses_queued": 5,
          "cves_found": 12,
          "exploitation_plan": [
            {
              "priority": 1,
              "cve_id": "CVE-2021-41773",
              "product": "Apache/2.4.49",
              "severity": "CRITICAL",
              "vuln_class": "path_traversal",
              "hypothesis_id": "H-0042",
              "attack_surface": "https://example.com/cgi-bin/.%2e/.%2e/etc/passwd",
              "exploit_available": true,
              "exploit_type": "metasploit",
              "exploit_url": "https://www.exploit-db.com/exploits/50383",
              "confidence": "HIGH",
              "recommended_payloads": ["../.../../etc/passwd", ".%2e/.%2e/etc/passwd"],
              "recommended_action": "Deploy Metasploit module immediately - CRITICAL severity RCE"
            },
            ...
          ],
          "summary": "Queued 5 CVE exploits: 2 CRITICAL, 3 HIGH",
          "status": "success"
        }
      </description>
      
      <status_codes>
        <code value="success">Hypotheses successfully queued</code>
        <code value="no_cves_found">No CVEs found matching criteria</code>
        <code value="error">CVE auto-integration failed</code>
      </status_codes>
    </returns>
    
    <example>
      <scenario>Apache 2.4.49 with known path traversal CVE</scenario>
      <input>
        tech_stack = {
          "web_servers": [
            {"name": "Apache", "version": "2.4.49", "confidence": "high"}
          ]
        }
        base_url = "https://target.com"
        min_severity = "HIGH"
      </input>
      <output>
        {
          "hypotheses_queued": 2,
          "cves_found": 2,
          "exploitation_plan": [
            {
              "priority": 1,
              "cve_id": "CVE-2021-41773",
              "severity": "CRITICAL",
              "vuln_class": "path_traversal",
              "attack_surface": "https://target.com/cgi-bin/.%2e/.%2e/etc/passwd",
              "exploit_available": true,
              "exploit_type": "metasploit"
            }
          ]
        }
      </output>
    </example>
    
    <example>
      <scenario>WordPress 5.8.2 with multiple CVEs</scenario>
      <input>
        tech_stack = {
          "cms": [
            {"name": "WordPress", "version": "5.8.2"}
          ]
        }
        base_url = "https://blog.example.com"
        min_severity = "MEDIUM"
        max_hypotheses = 10
      </input>
      <output>
        {
          "hypotheses_queued": 10,
          "cves_found": 15,
          "summary": "Queued 10 CVE exploits: 1 CRITICAL, 4 HIGH, 5 MEDIUM"
        }
      </output>
    </example>
    
    <security_notes>
      <note>This function ONLY queries public CVE databases - no interaction with target</note>
      <note>Hypothesis queueing is local - does not send payloads to target</note>
      <note>Rate limiting applied to NVD/ExploitDB queries (inherited from version_to_cves)</note>
    </security_notes>
    
    <integration_notes>
      <note>Designed to be called immediately after identify_tech_stack()</note>
      <note>Works seamlessly with existing hypothesis_ledger and payload generation</note>
      <note>Exploitation plan can be fed directly to agent's attack loop</note>
    </integration_notes>
  </tool>

=== enrich_hypothesis_with_cve ===
<tool name="enrich_hypothesis_with_cve">
    <description>Enrich an existing hypothesis with CVE metadata.
